The model and price of the printer of the highest price of each maker

SELECT product.maker, printer.model, printer.price
FROM printer, product, (SELECT maker, max(price) as price
             FROM printer
             GROUP BY maker) as exp
WHERE printer.model=product.model
    AND printer.price=exp.price
    AND product.maker=exp.maker;
